```python
def calculate_personality_type(scores):
    """
    scores: dict of 8 dimension scores (0-10)
    returns: tuple of (type_code, continuous_values, description)
    """
    # 计算三个维度
    A_score = scores['theory']
    C_score = scores['engineering']
    is_abstract = A_score > C_score

    B_score = (scores['radar'] + scores['learning']) / 2
    D_score = (scores['theory'] + scores['engineering']) / 2
    is_broad = B_score > D_score

    I_score = scores['innovation']
    O_score = (scores['collaboration'] + scores['influence']) / 2
    is_independent = I_score > O_score

    # 构建类型代码
    type_code = ('A' if is_abstract else 'C') + \
                ('B' if is_broad else 'D') + \
                ('I' if is_independent else 'O')

    # 计算连续值（用于气泡图位置）
    continuous = {
        'abstractness': A_score / 10,  # 0-1，越高越抽象
        'breadth_score': B_score / 10,  # 0-1，越高越广
        'independence': I_score / 10   # 0-1，越高越独立
    }

    return type_code, continuous
```
